def initialize(func, input = Queue.new, output = Queue.new)
        @input    = input
        @output   = output
        @handling = false

        @thread = Thread.new {
                while true
                        value = @input.deq

                        @handling = true
                        begin
                                value = func.call(value)
                                @output.enq value
                        rescue Exception; end
                        @handling = false
                end
        }
end
