def deliver(value)
	return self if delivered?

	@mutex.synchronize {
		@value = value

		cond.broadcast if cond?
	}

	self
end

def value(timeout = nil)
	return @value if delivered?

	@mutex.synchronize {
		cond.wait(@mutex, *timeout)
	}

	return @value if delivered?
end
