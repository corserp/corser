def test_treat_blank_arg_as_empty_string
  name, args = @app.parse_task_string("name[one,]")
  assert_equal "name", name
  assert_equal ["one", ""], args

  name, args = @app.parse_task_string("name[one,,two]")
  assert_equal "name", name
  assert_equal ["one", "", "two"], args
end
